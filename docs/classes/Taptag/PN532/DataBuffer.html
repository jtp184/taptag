def reset(arry = Array.new(255, 0))
  arry ||= Array.new(255, 0)
  ry = Array.new(255, 0)
            .map
            .with_index { |_i, x| i = arry[x] || 0 }

  to_ptr.write_array_of_uint8(ry)

  self
end
